# Правила фильтрации сообщений
#
# Формат правил:
#   field = pattern   — для текстовых полей (subject, from, folder)
#   field > number    — для числового поля size (больше)
#   field < number    — для числового поля size (меньше)
#
# Поддерживаемые поля:
#   subject — тема письма
#   from    — отправитель
#   folder  — имя папки
#   size    — размер сообщения в байтах
#
# Подстановочный символ * (wildcard) для текстовых полей:
#   текст*   — значение начинается на "текст"
#   *текст   — значение заканчивается на "текст"
#   *текст*  — значение содержит "текст"
#   текст    — точное совпадение
#
# Суффиксы размера (после числа, без пробела):
#   K или К — килобайты (умножение на 1024)
#   M или М — мегабайты (умножение на 1024×1024)
#   без суффикса — байты
#
# Строки, начинающиеся с #, являются комментариями.
# Строки с неверным форматом автоматически пропускаются.
#
# Примеры:
# subject = привет*
# subject = *тест*
# from = *ivan*
# from = *@mail.ru
# folder = INBOX
# size > 100K
# size < 5M

#subject = Retrieval using the IMAP4*
#from = info@calendar.yandex.ru
